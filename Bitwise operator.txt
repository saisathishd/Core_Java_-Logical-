//bitwise operator
import java.lang.*;
class bitwise{
    public static void main(String args[]){
        int a=2,b=3;
        System.out.println("Bitwise AND :"+(a&b));
        System.out.println("Bitwise OR :"+(a|b));
        System.out.println("Bitwise XOR :"+(a^b));
        System.out.println("Bitwise NOT :"+(~a));
    }
}
//o/p:
Bitwise AND :2
Bitwise OR :3
Bitwise XOR :1
Bitwise NOT :-3
